反转链表：

public ListNode reverse(ListNode head) {
    ListNode prev = null;
    ListNode curr = head;
    while (curr != null) {
        ListNode temp = curr.next;
        curr.next = prev;
        prev = curr;
        curr = temp;
    }
    return prev;
}

链表中点：

public ListNode mid(ListNode head) {
    ListNode slow = head;
    ListNode fast = head;
    // 什么时候用哑节点：头节点有可能被删除的时候
    while (fast != null) {
        slow = slow.next;
        fast = fast.next.next;
    }
    return slow;
}

链表中点的上一个（上半部分的最后一个）：

public ListNode mid(ListNode head) {
    ListNode slow = head;
    ListNode fast = head;
    while (fast.next != null && fast.next.next != null) {
        // 少运行一次
        slow = slow.next;
        fast = fast.next.next;
    }
    return slow;
}

合并两个有序链表（lc 21）：

ListNode mergeLists(ListNode l1, ListNode l2) {
    if (l1 == null) {
        return l2;
    } else if (l2 == null) {
        return l1;
    } else if (l1.val < l2.val) {
        l1.next = mergeList(l1.next, l2);
        return l1;
    } else {
        l2.next = mergeList(l1, l2.next);
        return l2;
    }
}

public void backTrack(TreeNode root) {
    if (root == null) {
        return;
    }
    res.add(root.val);
    backTrack(root.left);
    backTrack(root.right);
}

层序遍历

public List<List<Integer>> levelTrav(TreeNode root) {
    List<List<Integer>> res = new ArrayList<List<Integer>>();
    Queue<TreeNode> queue = new ArrayDeque<TreeNode>();
    if (root != null) {
        queue.offer(root);
    }
    while (!queue.isEmpty()) {
        int n = queue.size();
        List<Integer> level = new ArrayList<Integer>();
        for (int i = 0; i < n; i++) {
            TreeNode node = queue.poll();
            level.add(node.val);
            if (node.left != null) {
                queue.offer(node.left);
            }
            if (node.right != null) {
                queue.offer(node.right);
            }
        }
        res.add(level);
    }
    return res;
}

字典树#

class Trie {
    class TrieNode {
        boolean end;
        boolean[] next = new TrieNode[26];
    }
    
    TireNode root;
    public Trie() {
        root = new TrieNode();
    }
    
    public void insert(String word) {
        TrieNode p = root;
        for (int i = 0; i < word.length(); i++) {
            int cur = word.charAt(i) - 'a';
            if (p.next[cur] == null) {
                p.next[cur] = new TrieNode();
            }
            p = p.next[cur];
        }
        p.end = true;
    }
    
    public boolean search(String word) {
        TrieNode p = root;
        for (int i = 0; i < word.length(); i++) {
            int cur = word.charAt(i) - 'a';
            if (p.next[cur] == null) {
                return false;
            }
            p = p.next[cur];
        }
        return p.end;
    }
}

二分法#

public int search(int[] nums, int target) {
    int left = 0;
    int right = nums.length - 1;
    while (left <= right) {	// 左闭右闭这里用<=
        int mid = left + (right - left) / 2;
        if (nums[mid] < target) {
            left = mid + 1;
            // 由于上一行已经考虑了mid的值是不合适的，因此新的区间不能再包含mid
        } else if (nums[mid] > target) {
            right = mid - 1;
        } else if (nums[mid] == target) {
            return mid;
        }
    }
    return -1;
}

质数#

public boolean isPrime(int n) {
    if (n == 1) {
        return false;
    }
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            return false;
        }
    }
    return true;
}

堆排序#

// 一般用于寻找第k大的数
public int findKLargest(int[] nums, int k) {
    int n = nums.length;
    // 默认小根堆
    PriorityQueue<Integer> heap = new PriorityQueue<Integer>();
    // 堆的大小为k，先把nums的前k个元素加入heap中
    for (int i = 0; i < k; i++) {
        heap.offer(nums[i]);
    }
    // 考虑剩余的元素
    for (int i = k; i < n; i++) {
        Integer temp = heap.peek();	// 当前第k大的元素
        // 说明有更大的元素进来，要往后排队
        if (nums[i] > temp) {
            heap.poll();
            heap.offer(nums[i]);
        }
    }
    return heap.peek();
}

利用int判断String表示的数字是否溢出#

String s = "222222";
int res = 0;	// 要转换成的int
int sign = 1;	// 符号
int max = Integer.MAX_VALUE;
int min = Integer.MIN_VALUE;
for (int i = 0; i < s.length(); i++) {
    char cur = s.charAt(i);
    if (cur >= '0' && cur <= '9') {
        int num = cur - '0';
        if (res > max / 10 || (res == max / 10 && num > max % 10)) {
            // 除去最后一位的res > 除去最后一位的max
            // 或者除去最后一位的res == 除去最后一位的max且最后一位 > max的最后一位
            return max;
        } else if (res < min / 10 || (res == min / 10) && -num < max % 10) {
            return min;
        }
        res = res * 10 + sign * num;
    }
}

static final int near(int a) {
    int n = a - 1;
    n |= n >>> 1;	
    n |= n >>> 2;
    n |= n >>> 4;
    n |= n >>> 8;
    n |= n >>> 16;
    return (n < 0) ? 1 : n + 1;
}
